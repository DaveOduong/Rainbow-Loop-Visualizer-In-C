<script>

const boxes = document.querySelectorAll(".box");

const rainbow = [
    "red",
    "orange",
    "yellow",
    "green",
    "blue",
    "indigo",
    "violet"
];

let current = 0;

let running = true;

function animateBoxes(){

    // stop animation if paused
    if(!running){
        return;
    }

    // LOOP resets boxes
    for(let i = 0; i < boxes.length; i++){

        boxes[i].style.background = "gray";

        boxes[i].style.transform = "scale(1)";
    }

    // active rainbow color
    boxes[current].style.background =
        rainbow[current];

    // blinking effect
    boxes[current].style.transform =
        "scale(1.3)";

    current++;

    // restart animation
    if(current >= boxes.length){
        current = 0;
    }
}

// pause / continue button
function toggleAnimation(){

    running = !running;

    const btn = document.getElementById("btn");

    if(running){
        btn.innerText = "Pause";
    }
    else{
        btn.innerText = "Continue";
    }
}

setInterval(animateBoxes, 300);

</script>
